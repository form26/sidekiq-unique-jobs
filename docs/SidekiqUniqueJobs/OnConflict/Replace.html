  delete_lock
  block&.call

def delete_job_by_digest
  Scripts.call(
    :delete_job_by_digest,
    nil,
    keys: ["#{QUEUE_KEY}:#{queue}", SCHEDULE_SET, RETRY_SET], argv: [unique_digest],
  )
end
